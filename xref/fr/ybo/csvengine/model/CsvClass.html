17  package fr.ybo.csvengine.model;
18  
19  import java.lang.reflect.Constructor;
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  import fr.ybo.csvengine.exception.CsvEngineException;
24  
25  /**
26   * Represents a class associate with a CSV File.<br/><br/>
27   * <i><u>French :</i> Répésente une classe associé à un fichier CSV.
28   *
29   * @author ybonnel
30   */
31  public class CsvClass {
32  
33      /**
34       * Separator.
35       */
36      private final String separator;
37  
38      /**
39       * The class.
40       */
41      private final Class<?> clazz;
42  
43      /**
44       * The constructor.
45       */
46      private Constructor<?> constructor;
47  
48      /**
49       * Map of the filed in the class.
50       */
51      private final Map<String, CsvField> mapOfFields = new HashMap<String, CsvField>();
52  
53      /**
54       * Map of orders.
55       */
56      private final Map<String, Integer> orders = new HashMap<String, Integer>();
57  
58      /**
59       * Constructor.
60       *
61       * @param separator separator.
62       * @param clazz     the class.
63       */
64      public CsvClass(String separator, Class<?> clazz) {
65          this.separator = separator;
66          this.clazz = clazz;
67          try {
68              constructor = clazz.getDeclaredConstructor((Class<?>[]) null);
69              constructor.setAccessible(true);
70          } catch (Exception e) {
71              throw new CsvEngineException("Error while getting the default constructor of " + clazz.getSimpleName()
72                      + ", this class miss a constructor without parameters", e);
73          }
74      }
75  
76      /**
77       * Get the CsvField associated with the column name.
78       *
79       * @param columnName name of column.
80       * @return the field.
81       */
82      public CsvField getCsvField(String columnName) {
83          return mapOfFields.get(columnName);
84      }
85  
86      /**
87       * @return the list of column names.
88       */
89      public Iterable<String> getColumnNames() {
90          return mapOfFields.keySet();
91      }
92  
93      /**
94       * @return the class.
95       */
96      public Class<?> getClazz() {
97          return clazz;
98      }
99  
100     /**
101      * @return the constructor.
102      */
103     public Constructor<?> getConstructor() {
104         return constructor;
105     }
106 
107     /**
108      * Separator without escape.
109      */
110     private String separatorWithoutEscape = null;
111 
112     /**
113      * @return Separator without escape.
114      */
115     public char getSeparatorWithoutEscape() {
116         if (separatorWithoutEscape == null) {
117             separatorWithoutEscape = separator.replaceAll("\\\\", "");
118         }
119         if (separatorWithoutEscape.length() != 1) {
120             throw new CsvEngineException("The separator " + separatorWithoutEscape
121                     + " contains more than 1 character.");
122         }
123         return separatorWithoutEscape.charAt(0);
124 
125     }
126 
127     /**
128      * @param columnName name of the column.
129      * @param csvField   CSV Field.
130      */
131     public void setCsvField(String columnName, CsvField csvField) {
132         mapOfFields.put(columnName, csvField);
133     }
134 
135     /**
136      * Add of the order.
137      *
138      * @param columnName name of the column.
139      * @param order      order.
140      */
141     public void putOrder(String columnName, int order) {
142         orders.put(columnName, order);
143     }
144 
145     /**
146      * Get the order of a field.
147      *
148      * @param columnName name of the column.
149      * @return the order.
150      */
151     public int getOrder(String columnName) {
152         return orders.get(columnName);
153     }
154 
155 }
